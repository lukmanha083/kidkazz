# Credit Card Expense Workflow - Automatic Journal Entry

## Overview

This document defines the workflow for handling business expenses paid via company credit card. Common use cases include:

- Cloud services (AWS, Google Cloud, DigitalOcean)
- Software subscriptions (GitHub, Figma, Slack, Notion)
- Development services (Upwork, freelancers)
- Domain & hosting
- Online advertising (Google Ads, Facebook Ads)
- Other recurring business expenses

---

## Database Schema

### Credit Card Transactions Table

```sql
CREATE TABLE credit_card_transactions (
  id TEXT PRIMARY KEY,
  transaction_number TEXT UNIQUE NOT NULL,  -- CCE-2025-0001

  -- Credit Card
  credit_card_account_id TEXT NOT NULL REFERENCES chart_of_accounts(id),

  -- Transaction Details
  transaction_date INTEGER NOT NULL,
  merchant_name TEXT NOT NULL,
  description TEXT,

  -- Amount
  original_currency TEXT DEFAULT 'IDR',
  original_amount REAL,
  exchange_rate REAL DEFAULT 1,
  amount_idr REAL NOT NULL,

  -- Expense Categorization
  expense_account_id TEXT NOT NULL REFERENCES chart_of_accounts(id),
  expense_category TEXT,

  -- Recurring
  is_recurring INTEGER DEFAULT 0,
  recurring_frequency TEXT CHECK(recurring_frequency IN ('WEEKLY', 'MONTHLY', 'QUARTERLY', 'YEARLY')),

  -- Journal Entry Reference
  journal_entry_id TEXT REFERENCES journal_entries(id),

  -- Reconciliation
  statement_period TEXT,  -- 2025-01
  is_reconciled INTEGER DEFAULT 0,
  reconciled_at INTEGER,

  -- Status
  status TEXT DEFAULT 'POSTED' CHECK(status IN ('DRAFT', 'POSTED', 'VOIDED')),

  -- Audit
  created_at INTEGER NOT NULL,
  created_by TEXT,
  updated_at INTEGER NOT NULL,

  FOREIGN KEY (credit_card_account_id) REFERENCES chart_of_accounts(id),
  FOREIGN KEY (expense_account_id) REFERENCES chart_of_accounts(id)
);

CREATE TABLE credit_card_payments (
  id TEXT PRIMARY KEY,
  payment_number TEXT UNIQUE NOT NULL,  -- CCP-2025-0001

  -- Credit Card & Bank
  credit_card_account_id TEXT NOT NULL REFERENCES chart_of_accounts(id),
  bank_account_id TEXT NOT NULL REFERENCES chart_of_accounts(id),

  -- Payment Details
  payment_date INTEGER NOT NULL,
  statement_period TEXT NOT NULL,  -- 2025-01
  statement_amount REAL NOT NULL,

  -- Payment
  payment_type TEXT NOT NULL CHECK(payment_type IN ('FULL', 'PARTIAL', 'MINIMUM')),
  principal_amount REAL NOT NULL,
  interest_amount REAL DEFAULT 0,
  total_payment REAL NOT NULL,

  -- Journal Entry Reference
  journal_entry_id TEXT REFERENCES journal_entries(id),

  -- Status
  status TEXT DEFAULT 'POSTED' CHECK(status IN ('DRAFT', 'POSTED', 'VOIDED')),

  -- Audit
  created_at INTEGER NOT NULL,
  created_by TEXT,

  FOREIGN KEY (credit_card_account_id) REFERENCES chart_of_accounts(id),
  FOREIGN KEY (bank_account_id) REFERENCES chart_of_accounts(id)
);
```
